// c
#include <stdio.h>
typedef struct _user
{  
    char name[64];
    int  age;  
} user;  
int main(int argc, char *argv[])
{
    user user = {
        .name="jack",
        .age=20
    };
    printf("%s\n", user.name);
    printf("%d\n", user.age);
    return 0;
}

// rust
struct User {
    name: String,
    age: i32,
}
fn main() {
    let user = User {
        name: String::from("jack"),
        age: 32
    };
    println!("{} {}", user.name, user.age);
}
