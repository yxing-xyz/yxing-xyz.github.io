cd /usr/share/kbd/keymaps/i386/qwerty
# 解压键盘map.gz
gzip -d us.map.gz
# 编辑文件修改：
vim us.map
# Caps改为Ctrl
keycode  58 = Control
keycode  86 = less             greater          bar
keycode  97 = Control
Keycode  100 = Caps_Lock
# 重新压缩成.gz
gzip us.map
